                           (token.value[0],
                            token.lexer.lineno,
                            token.lexer.lexpos - self._previous_newline_position))

[docs]    def _begin_block(self, token, state):
        self._logger.debug('')
        token.lexer.code_start = token.lexer.lexpos
        token.lexer.level = 1 # Initial brace level
        token.lexer.push_state(state)

[docs]    def _decrement_block_level(self, token, token_type):
        self._logger.debug('')
        token.lexer.level -=1
        if token.lexer.level == 0:
            token.type = token_type
            token.value = token.lexer.lexdata[token.lexer.code_start:token.lexer.lexpos-1]
            token.lexer.pop_state()
            return token

[docs]    def _begin_identified_block(self, token, state):
        self._logger.debug('')
        token.value = token.value[1:-1]
        if token.value.lower() == 'string':
            token.type = 'StringDefinition'
        else:
            token.type = 'Entry'
        token.lexer.push_state(state)



[docs]    def _end_identified_block(self, token):
        self._logger.debug('')
        token.type = 'EndOfBlock'
        token.lexer.pop_state()

[docs]    def p_key_value_pair_sequence(self, p):
        ''' key_value_pair_sequence : key_value_pair COMMA key_value_pair_sequence
                                    | key_value_pair COMMA
                                    | key_value_pair '''
        self._logger.debug('')
        if len(p) == 4:
            pairs, pair = p[3], p[1]
            pairs.append(pair)
            p[0] = pairs
        else:
            pair = p[1]
            p[0] = [pair]

[docs]    def _build(self, **kwargs):

        self._lexer = lex.lex(module=self, **kwargs)
        self._parser = yacc.yacc(module=self, **kwargs)

[docs]    def _reset(self):

        self._previous_newline_position = 0

[docs]    def parse(self, text):

        self._reset()
        self._parser.parse(text, lexer=self._lexer)
