    def column_names(cls, only_known=False):

        mapper = inspect(cls)
        # Wrong: return mapper.attrs.keys()
        if only_known:
            return [column.name for column in mapper.columns
                    if 'unknown' in column.info and column.info['unknown']]
        else:
            return mapper.columns.keys()
